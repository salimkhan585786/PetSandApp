// Triggered when order.status changes in Firestore ( cloud site )
const statusMessages = {
  packed: 'Your order is packed and ready!',
  out_for_delivery: 'Your sand is on the way! 📨',
  delivered: 'Delivered! Enjoy fresh litter 🐾',
  cancelled: 'Your order was cancelled. Refund in 5-7 days.',
};
exports.onOrderStatusChange = functions.firestore
  .document('orders/{orderId}')
  .onUpdate(async (change, context) => {
    const { status, userId, fcmToken } = change.after.data();
    const msg = statusMessages[status];
    if (!msg || !fcmToken) return;
    await admin.messaging().send({
      token: fcmToken,
      notification: { title: 'PetSand Update', body: msg },
      data: { orderId: context.params.orderId, status },
    });
  });
